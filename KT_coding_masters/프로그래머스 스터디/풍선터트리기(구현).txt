# 최후에 남는 풍선 특
# 1 . 양쪽 끝에 위치 , 
# 2. 왼쪽과 오른쪽 둘다 큰값이 존재 
# 3. 왼쪽은 모두 크고 오른쪽은 모두 작음 or 오른쪽은 모두 크고 왼쪽은 모두 작음

# >> 자신의 왼쪽 혹은 오른쪽 어느 한 쪽 방향에 자기보다 큰 수만 존재할 시에 마지막에 남기는게 가능함
def solution(a):
    result = [False for _ in range(a)]
    minFront, minRear = float("inf"), float("inf")
    for i in range(len(a)):
        if a[i] < minFront:
            minFront = a[i]
            result[i] = True
        if a[-1-i] < minRear:
            minRear = a[-1-i]
            result[-1-i] = True
    return sum(result)
